import json
import pandas as pd

# Baca file JSON
with open("Zam Zam Zahrina_V3925037_Literasi3-2.json", "r", encoding="utf-8") as f:
    data = json.load(f)

# Konversi setiap bagian JSON ke DataFrame
df_laporan_peserta = pd.DataFrame(data["laporan_peserta_demonstran"])
df_gps = pd.DataFrame(data["data_Lokasi_gps"])
df_tim_medis = pd.DataFrame(data["basis_data_tim_medis"])
df_laporan_warga = pd.DataFrame(data["laporan_warga_sekitar"])

# Simpan ke file Excel dengan beberapa sheet
with pd.ExcelWriter("output_data.xlsx", engine="openpyxl") as writer:
    df_laporan_peserta.to_excel(writer, sheet_name="Peserta Demonstran", index=False)
    df_gps.to_excel(writer, sheet_name="Data GPS", index=False)
    df_tim_medis.to_excel(writer, sheet_name="Tim Medis", index=False)
    df_laporan_warga.to_excel(writer, sheet_name="Laporan Warga", index=False)

print("Konversi JSON ke XLSX selesai. File tersimpan sebagai output_data.xlsx")
